        def generate(str2encode, logopng)

                # get the qr code matrix for this input string with "high" redundancy so we can embed a logo
                # note: don't specify :size, let it figure out which "version" to use for us (version determines 
                # number of modules/squares)
                qrCodeMatrix = RQRCode::QRCode.new(str2encode, :level => :h )
                
                image_width = self.image_width_pixels qrCodeMatrix.module_count
                image_height = image_width

                transparent_img = Image.new(image_width,image_height) { self.background_color = "transparent" }




                return transparent_img
        
        end
